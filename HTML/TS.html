# 9、函数

function foo(a, b) {
    return a + b
}
// Parameter 'a' implicitly has an 'any' type.

// 给参数分配一个数据类型
function foo(a: number, b: string) {
    return a + b
}
// 给返回值做分配
// void表示没有返回值
function foo(a: number, b: string): void {
    // ...
}

// 可选参数，可选参数应该在所有参数的后面
function foo(a: number, b?: number): number {
    return a + b
}

// 剩余参数,rest是一个数组结构，需要指定数组类型
function foo(a: number, b?: number, ...rest: string[]): number {
    return a + b
}

const f = foo(20, 30, 'a')

# 10、接口

关键字 interface ,通常定义一个对象

interface Obj {
    name: string,
    age: number
}

const obj: Obj {
    name: 'a',
    age: 10
}

// 避免出现数据类型和键的书写错误
const obj: Obj {
    name: 'a',
    ago: 10
}


interface Obj {
    name: string,
    age: number
}

const obj: Obj = {
    name: 'a',
    ago: 10
}

// Object literal may only specify known properties, and 'ago' does not exist in type 'Obj'.

# 3、类的修饰符

  • 可选属性
  • 属性默认值
  • 私有属性
  • 受保护属性
  • 静态成员
class Article {
    // 设置属性默认为public
    title: string
    content: string
    // 设置可选属性使用问号

    // 设置属性默认值
    views = 100
    
    // 私有属性，在子类中是无法使用的
    private subtitle?: string
    // 受保护的属性，只能在Article和其子类中进行访问
    protected thumbs: number
    
    // 静态属性
    static subs: number
    // 私有静态属性, 不可以被子类继承，不可以在被外部调用
   
    private static shortTitle: string

    private static readonly brief = 'abc'
    
    // 构造函数，title和content是必选属性
    constructor (title: string, content: string) {
        this.title = title
        this.content = content
        // 调用静态私有属性
        // Article.shortTitle
    }
}

抽象的成员必须在抽象类中才能使用

abstract class Animal {
    // 抽象属性
    abstract name: string,
    // 抽象方法
    abstract makeSound(): void
    // 非抽象方法

    }
}

class Cat extends Animal {

# 7、泛型类

在类的名称后面加一个<T>

class MyClass<T> {
    value: T
    
    constructor (value: T) {
        this.value = value
    }
    
    do (param: T) {
        console.log('I can do', this.value)
    }
}

// 如果传的是一个字符串
const myStr = new MyClass<string>('hello')
myStr.do('cry')

// 如果传的是一个数字
const myNum = new MyClass<number>(10)
myNum.do(10)
