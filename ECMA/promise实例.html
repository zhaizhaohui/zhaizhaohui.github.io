# promise实例

# 1、异步读取文件

const fs = require('fs').promises;

function readFileAsync(filePath) {
    return new Promise((resolve, reject) => {
        fs.readFile(filePath, 'utf8')
          .then((data) => {
                resolve(data);
            })
          .catch((err) => {
                reject(err);
            });
    });
}

// 使用示例
readFileAsync('example.txt')
  .then((data) => {
        console.log('文件内容：', data);
    })
  .catch((err) => {
        console.error('读取文件出错：', err);
    });

首先通过 fs.promises.readFile 函数来异步读取文件。

我们将Promise包装在一个自定义的 readFileAsync 函数中，该函数返回一个 Promise 对象。

在 Promise 的构造函数中，我们通过调用 fs.readFile 并根据其结果来决定是调用 resolve（成功时）还是 reject（失败时）。

最后，我们使用 then 来处理读取成功的情况，用 catch 来处理读取失败的情况。

# 2、多个异步操作依次执行

// 模拟获取用户信息的异步函数
function getUserInfoAsync() {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            const userInfo = {
                id: 1,
                name: 'John Doe',
            };
            resolve(userInfo);
        }, 1000);
    });
}

// 模拟根据用户信息获取订单列表的异步函数
function getOrderListAsync(userInfo) {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            const orderList = [
                {
                    id: 101,
                    amount: 100,
                },
                {
                    id: 102,
                    amount: 200,
                },
            ];
            resolve(orderList);
        }, 1000);
    });
}

// 模拟根据订单列表获取订单详情的异步函数
function getOrderDetailsAsync(orderList) {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            const orderDetails = orderList.map((order) => ({
               ...order,
                details: 'Some details about the order',
            }));
            resolve(orderDetails);
        }, 1000);
    });
}

// 依次执行三个异步操作
getUserInfoAsync()
  .then((userInfo) => {
        return getOrderListAsync(userInfo);
    })
  .then((orderList) => {
        return getOrderDetailsAsync(orderList);
    })
  .then((orderDetails) => {
        console.log('订单详情：', orderDetails);
    })
  .catch((err) => {
        console.error('出错：', err);
    });

在这个实例中，首先定义了三个模拟不同异步操作的函数，每个函数都返回一个 Promise 对象。

然后，我们通过链式调用 then 函数，将每个异步操作依次连接起来，使得每个操作都能获取到前一个操作的结果，并最终在所有操作成功完成后，在最后一个 then 分支中处理订单详情的结果。

如果任何一个操作出现错误，都会进入 catch 分支进行错误处理。

# 3、并行执行多个异步操作

假设我们要同时获取多个不同的 API 数据，比如同时获取天气数据和新闻数据。

// 模拟获取天气数据的异步函数
function getWeatherDataAsync() {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            const weatherData = {
                temperature: 25,
                condition: 'Sunny',
            };
            resolve(weatherData);
        }, 1000);
    });
}

// 模拟获取新闻数据的异步函数
function getNewsDataAsync() {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            const newsData = [
                {
                    title: 'Breaking News 1',
                    content: 'Some news content 1',
                },
                {
                    title: 'Breaking News 2',
                    content: 'Some news content 2',
                    },
            ];
            resolve(newsData);
        }, 1000);
    });
}

// 并行执行获取天气数据和新闻数据的异步操作
Promise.all([getWeatherDataAsync(), getNewsDataAsync()])
  .then(([weatherData, newsData]) => {
        console.log('天气数据：', weatherData);
        console.log('新闻数据：', newsData);
    })
  .catch((err) => {
        console.error('出错：', err);
    });

在这个实例中，定义了两个模拟获取不同数据的异步函数，它们都返回一个 Promise 对象。

然后，我们使用 Promise.all 函数来并行执行这两个异步操作。当所有的异步操作都成功完成后，Promise.all 的 then 分支会接收到一个包含所有操作结果的数组，我们可以按照顺序从数组中提取出每个操作的结果并进行处理。

如果任何一个操作出现错误，都会进入 catch 分支进行错误处理。

# 4、用户登录

模拟用户登录后获取用户信息

let status = 1
// 用户登录
let login = (resovle, reject) => {
    if (status === 1) {
        resovle({data: '', msg: '登录成功'})
    }else{
        reject('登录成功')
    }
}
// 获取用户信息
let getInfo = (resovle, reject) => {
    if (status === 1) {
        resovle({data: '', msg: '获取成功'})
    }else{
        reject('获取成功')
    }
}
// 使用Promise进行判断操作
new Promise(login).then(res => {
    console.log(res)   // 登录成功
    // 登录成功之后获取用户信息
    return new Promise(getInfo)
}).then(res => {
    console.log(res)   // 获取成功
})
