# Proxy 代理

Proxy用于修改某些操作的默认行为，也可以理解为在目标对象之前架设一层拦截，外部所有的访问都必须先通过这层拦截，因此提供了一种机制，可以对外部的访问进行过滤和修改。 这个词的原理为代理，在这里可以表示由它来“代理”某些操作，译为“代理器”。

作用：比如vue中的拦截、预警、上报、扩展功能、统计、增强对象等

# 语法

new Proxy(target, hanlder)
// target->被代理的对象
// handler->对被代理对象的操作(json格式)

// Example
let obj = {}
new Proxy(obj, {
    // 设置内容时
    set () {
        
    },
    // 获取内容是
    get () {
        
    },
    // 调用函数时
    apply () {
        
    },
    // 查询是否存在时，如使用 xx in obj 会调用
    has() {
        
    },
    // 删除时
    deleteProperty() {
        
    },
    // .....
})

# 实战案例

现在访问一个对象的属性，如果不存在的时候给出了undefined，希望如果不存在给出错误（警告）信息

let obj  = {
    name: 'king'
}
let newObj = newProxy(obj, {
    get(target, property) {
        // property为外部访问的属性
        if (property in target) {
            return target[property]
        } else {
            throw new ReferenceError(`${property}属性不存在与对象上`)
            console.warn(`${property}属性不存在与对象上`)
        }
    }
})

console.log(newObj.name)
console.log(newObj.age)

# 实战案例2

创建DOM

const DOM = new Proxy({}, {
    get(target, property) {
        return function(attr = {}, ...children) {
            const el = document.createElement(property)
            // 添加属性
            for (let key of Object.keys(attr)) {
                el.setAttribute(key, attr[key])
            }
            // 添加子元素
            for (let child of children) {
                if (typeof child == 'string') {
                    child = document.createTextNode(child)
                } 
                el.appendChild(child)
            }
            return el
        }
    }
})

let oDiv = DOM.div({
    // 创建div
    {id: 1, class: 'z-ui'}, 'my name is king',
    // 创建链接
    DOM.a{herf: 'https://www.baidu.com'}, '百度一下')
    // 创建列表
    DOM.ul({},
        DOM.li({}, 'king'),
        DOM.li({}, 'sugar'),
        DOM.li({}, 'butter'),
    )
}) 

window.onload = function() {
    document.body.appendChild(oDiv)
}
