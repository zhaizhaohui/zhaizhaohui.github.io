# 解构赋值

解构赋值语法是一种JavaScript表达式用来将数组中的值或对象中的属性取出来区分为不同变量。

解构赋值非常有用，特别是做数据交互的时候。在解构的时候，需要注意的是，左右两边结构格式要一样。

# 解构数组

let [a, b, c] = [5, 6, 7]
console.log(a, b, c)
// 5, 6, 7

// 可以给元素默认值
let [a, b, c] = [5, 6]
console.log(a, b, c)
// 5, 6, undefined
let [a, b, c='暂无数据'] = [5, 6]
console.log(a, b, c)
// 5, 6, 暂无数据

// 如果后台传过来的是undefined，可以给默认值
let [a, b, c='暂无数据'] = [5, 6, undefined]
// 5, 6, 暂无数据

// 如果后台传过来的是null
let [a, b, c] = [5, 6, null]
console.log(a, b, c)
// 5, 6, null
// 如果需要null给默认值，需要在console的时候做判断

# 解构JSON

let json = {
    name: 'king',
    age: 18,
    job: 'worker'
}
let {name, age, job} = json
console.log(name, age, job)
// king, 18, worker

// 不需要完全对应
let {n, a, g} = json
console.log(n, a, g)
// king, 18, worker

// 可以使用别名,使用冒号:
let {name, age, job:g} = json
console.log(name, age, job)
// job is undefined
console.log(name, age, a)
// king, 18, worker

交换数的位置
let a = 5
let b = 6
[a, b] = [b, a]
console.log(a, b)
// 6, 5

# 解构函数返回值

function getPos () {
    return {
        top: 10,
        left: 20
    }
}
let {top:t, left:l} = getPos()
console.log(t,l)
// 10, 20
