# js

$(document).on('change', '#file', function(e){
    let file = $(this).prop('files')[0]
    let reader = new FileReader()
    // 将图片转成base64格式
    reader.readAsDataURL(file)
    // 读取成功后的回调
    reader.onloadend = function () {
        let img = new Image()
        // 获取回调结果
        img.src = this.result
        img.onload = function(){
            let line = $('#line').val(),
                row = $('#row').val(),
                width = img.width + row * 5,
                height = img.height + line * 5
            console.log(line + '|' + row)
            $('#show').css({'width': width, 'height': height})
            demo(img, line, row)
        }
    }
})

function demo (img, line, row) {
    let h = img.height / line,
        w = img.width / row
    console.log(img.height)
    for (let i = 0; i < line; i++) {
        for (let r = 0; r < row; r++) {
            let    canvas = document.createElement('canvas'),
                context = canvas.getContext('2d')
            canvas.width = w
            canvas.height = h
            context.drawImage(img, r * w, i * h, w, h, 0, 0, w, h)
            let image = new Image()
            image.name = i
            image.src = canvas.toDataURL('image/jpeg')
            $('#show').append(image)
        }
    }
}
