# Javascript

// 定义函数
function jumpTo(n) {
    // 获取容器元素
	var divBox = document.getElementById('list')
     // 获取定位元素(使用索引)
    var liBox = document.getElementsByTagName('li')[n-1]
	 // 获取第一个li元素相对于父元素的距离顶端位置
     var top = document.getElementsByTagName('li')[0].offsetTop
	  // scrollTop：位于对象最顶端和窗口中可见内容的最顶端之间的距离
     // 简单地说就是滚动后被隐藏的高度
     divBox.scrollTop = liBox.offsetTop - top
}
// 执行函数
jumpTo(3)
