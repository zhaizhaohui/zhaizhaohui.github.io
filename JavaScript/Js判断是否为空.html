# Js判断是否为空

# 1、判断object对象

jQuery.isEmptyObject(object)
 // 为空返回true
 // 用于对象未指定类型，或指定类型为数组，对象三者时可用，其他类型不可用 

object == undefined
// 为空返回true 未指明类型

typeof(object ) == “undefined“
 // 为空返回true 未指明类型 

typeof(object ) != “object”
 // 为空返回true 未指明类型

!object
 // 为null,undefined,0则为true
 // 除了number类型可以用于任何类型，或者未指定类型
