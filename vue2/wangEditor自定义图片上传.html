# wangEditor自定义图片上传

# 1、创建编辑器组件

Editor.vue

<script>
	import E from 'wangeditor'
	import ImgList from './ImgList'
	export default {
		name: 'Editor',
		components: {
			ImgList
		},
		props: {
			content: {
				type: String,
				default () {
					return ''
				}
			}
		},
		mounted () {
			this.initEditor()
		},
		data () {
			return {
				imgs: ''
			}
		},
		methods: {
			// 显示图片选择层
			handleClick () {
				this.$refs.drawer.isShow = true
			},
			// 接受传递过来的图片链接，并插入到编辑器
			getImgs (imgs) {
				// 传递过来的是一个数组，数组转字符串
				this.imgs = imgs.join('|#|')
                            // 解决数据同步问题
				this.$nextTick(function () {
					document.getElementById('insertBtn').click()
				})
				// 隐藏选择层
				this.$refs.drawer.arr = []
				this.$refs.drawer.isShow = false
			},
			// 进行上传操作
			initEditor () {
				// 初始化编辑器
				let editor = new E('weditor')
				// 编辑器配置
				editor.config.menus = [
					'source',
					'|',
					'fontsize',
					'bold',
					'eraser',
					'forecolor',
					'|',
					'img',
					'link',
					'unlink',
					'table',
					'|',
					'alignleft',
					'aligncenter',
					'alignright',
					'|',
					'undo',
					'redo',
					'fullscreen'
				]
				// 编辑器渲染
				editor.create()
				// 编辑器高度自适应
				document.getElementById('weditor').setAttribute('style', 'height:auto;min-height:2rem')
			}
		}
	}
</script>

# 2、创建图片选择组件

ImgList.vue

<script>
	export default {
		name: 'ImgList',
		data () {
			return {
                           // 存储选择项
				arr: [],
                            // 默认图层为隐藏
				isShow: false,
                            // 保存所选图片的url地址
				getImgs: [],
                            // 获取到的图片集
                            imgList: []
			}
		},
		methods: {
			// 选择图片
			selectImg (i) {
				if (this.arr.includes(i)) {
					this.arr = this.arr.filter((ele) => {
						return ele !== i
					})
				} else {
					this.arr.push(i)
				}
			},
			// 将选中的图片地址传出
			insertImg () {
				// 获取图片链接地址
				for (let i = 0; i < this.arr.length; i++) {
                                this.getImgs.push(this.imgList[this.arr[i]])
				}
                           // 将获取的图片内容以数组的形式传递个父组件Editor.vue
				this.$emit('getImgs', this.getImgs)
				// 清空数组防止重复push
				this.getImgs = []
			}
		}
	}
</script>

# 3、修改wangEdit.js

修改wangEdit.js中图片插入菜单

本例中使用的时wangEditro2版本

// img 菜单
_e(function (E, $) {

    E.createMenu(function (check) {
        var menuId = 'img';
        if (!check(menuId)) {
            return;
        }
        var editor = this;
        var lang = editor.config.lang;

        // 创建 menu 对象
        var menu = new E.Menu({
            editor: editor,
            id: menuId,
            title: lang.img
        });
        // 插入图片
        $(document).on('click','#insertBtn',function(e){
            // 获取页面存储的图片值
            let src = $('#imgs').val();
            let imgHtml = ''
            // 判断字符串是否包含分隔符，用于多图插入
            if (src.indexOf('|#|') > -1) {
                // 分割成数组
                let arr = src.split('|#|')
                for (var i = 0; i < arr.length; i++) {
                    imgHtml += '<img style="max-width:100%;" src="'+arr[i]+'"/>'
                }
            }else {
                imgHtml = '<img style="max-width:100%;" src="'+src+'"/>'
            }
            editor.command(e, 'insertHtml', imgHtml);
        })
        // 点击事件
        menu.clickEvent = function (e) {
            var menu = this;
            var dropPanel = menu.dropPanel;
            // 在此打开图片选择页面
            document.getElementById('imgUpbtn').click() 
        }
        // 增加到editor对象中
        editor.menus[menuId] = menu;
    })
});
