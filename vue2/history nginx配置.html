# history nginx配置

# 1、在vue的router/index.js中

import Vue from 'vue'
import Router from 'vue-router'
Vue.use(Router)
export default new Router({
	// 开启history模式，去掉#
    mode: 'history',
    // 如果需要绑定二级目录，默认是/
    base: '/ac',
    // 路由
    routes: [
        ......
    ]
})

# 2、在Nginx中，配置nginx.conf文件

server {
    listen	80;
    server_name	xxx.com;
    root   /usr/local/nginx/html/demo;

     location ^~/ac {
        index index.html;
        try_files $uri $uri/ /ac/index.html;
    }

    location ^~/api/ {
        add_header 'Access-Control-Allow-Origin' '*';
        proxy_pass https://cv.178661.com/agent/;
    }
}
